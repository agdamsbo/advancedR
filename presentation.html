Code - Euler plot fatigue by sex

colnames(df)[1:5]<-c("general","physical","mental","motivation","activity")

fs <- list()
ls <- levels(factor(df$sex))
for (i in seq_along(ls)){
  fs[[i]] <- eulerr::euler(df|>filter(sex==ls[i])|>select(1:5),shape="ellipse")
  names(fs)[i]<-ls[i]
  }
ps<-list()

for (i in seq_along(ls)){
  ps[[i]] <- plot(fs[[i]],
     quantities = TRUE,
     fill = cs,
     lty = 1:5,
     labels = list(font = 4))
  names(ps)[i]<-ls[i]
}
